def _make_dm20_handler(rng: random.Random) -> "callable":
    """Return an async respx side-effect with deterministic 0-50ms variance."""
    async def handler(request: httpx.Request) -> httpx.Response:
        # Tiny sleep so scheduler can interleave coroutines (D-196).
        delay = rng.uniform(LATENCY_MIN_S, LATENCY_MAX_S)
        await asyncio.sleep(delay)
        # Parse body to route on tool_name; keep response shape minimal.
        import json
        body = json.loads(request.content.decode("utf-8") or "{}")
        tool = body.get("tool_name", "unknown")
        if tool == "dm20__get_class_info":
            return httpx.Response(200, json={"class": body["arguments"].get("name", "fighter"), "hd": "1d10"})
        if tool == "dnd__search_all_categories":
            return httpx.Response(200, json={"results": [{"name": body["arguments"].get("q", "")}]})
        # combat-ish bypass-only tool — never cached, exercises MCPClient hot path
        if tool == "dm20__combat_action":
            return httpx.Response(200, json={"result": "ok", "round": body["arguments"].get("round", 0)})
        return httpx.Response(200, json={"ok": True, "tool": tool})
    return handler
```

async def _channel_worker(
    channel_id: str,
    pc_ids: list[str],
    cache,              # MCPCache shared across channels
    wq,                 # WriterQueue shared
    channel_repo,       # ChannelSessionRepo shared
    view_repo,          # PersistentViewRepo shared
    memory_registry,    # MonsterMemoryRegistry shared
    errors: list[str],
    bypass_call_count: dict[str, int],
    cacheable_call_count: dict[str, int],
) -> None:
    """
    One simulated bot channel: 5 rounds, each round:
      - 3 cacheable lookups (mix of get_class_info + search_all_categories)
      - 1 bypass-only combat_action (exercises MCPClient hot path)
      - 1 channel-session upsert + 1 persistent-view insert (WriterQueue writes)
      - 4 monster-memory observations (one per PC, against shared monster_id)
      - SmartMonsterDriver-style cache-key write asserted later
    """
    from datetime import UTC, datetime
    from eldritch_dm.persistence.models import ChannelState, PersistentView

    try:
        for round_no in range(ROUNDS_PER_CHANNEL):
            # — Cacheable MCP reads (drives L1 hit/miss accounting) —
            for i in range(MONSTERS_PER_ROUND):
                # Same arg-set across all channels → first channel = MISS, rest = HIT.
                if i % 2 == 0:
                    await cache.call("dm20__get_class_info", name="fighter")
                else:
                    await cache.call("dnd__search_all_categories", q="fireball")
                cacheable_call_count[channel_id] = cacheable_call_count.get(channel_id, 0) + 1

            # — Bypass-only MCP call (combat action — never cacheable) —
            await cache.call("dm20__combat_action", round=round_no, channel=channel_id)
            bypass_call_count[channel_id] = bypass_call_count.get(channel_id, 0) + 1

            # — WriterQueue writes (channel state + view insert) —
            await channel_repo.upsert(
                channel_id=channel_id,
                campaign_name=f"Camp-{channel_id}",
                state=ChannelState.COMBAT,
            )
            await view_repo.insert(
                PersistentView(
                    custom_id=f"{channel_id}-r{round_no}",
                    view_class="CombatView",
                    message_id=f"msg-{channel_id}-{round_no}",
                    channel_id=channel_id,
                    payload={"round": round_no},
                    created_at=datetime.now(UTC),
                )
            )

            # — MonsterMemory observations (per-channel isolation test) —
            # Same monster_id across all channels — registry MUST keep them isolated.
            mem = memory_registry.recall(channel_id, "stress-session", "monster-x")
            for pc_id in pc_ids:
                mem.observe_hit(pc_id, damage=3 + round_no, round_number=round_no)

            # — Yield to let other coroutines interleave —
            await asyncio.sleep(0)
    except Exception as exc:  # noqa: BLE001
        errors.append(f"{channel_id}: {type(exc).__name__}: {exc}")
```

```python
@respx.mock
async def test_4_channel_concurrent_stress(tmp_path, monkeypatch):
    """The full 4-channel scenario. D-195 assertions (a)-(e) inline below."""
    from eldritch_dm.mcp.cache import MCPCache
    from eldritch_dm.mcp.client import MCPClient
    from eldritch_dm.gameplay.monster_memory import MonsterMemoryRegistry
    from eldritch_dm.persistence.bootstrap import bootstrap
    from eldritch_dm.persistence.channel_sessions_repo import ChannelSessionRepo
    from eldritch_dm.persistence.connection import WriterQueue, open_connection
    from eldritch_dm.persistence.persistent_views_repo import PersistentViewRepo

    # Enable L2 so assertion (a) also covers L2 SQLite WAL behavior.
    monkeypatch.setenv("MCPCACHE_ENABLED", "true")
    monkeypatch.setenv("MCPCACHE_L2_ENABLED", "true")
    monkeypatch.setenv("MCPCACHE_L2_PATH", str(tmp_path / "mcp_cache.sqlite"))

    # Hermetic dm20 mock (D-196).
    rng = random.Random(STRESS_RNG_SEED)
    respx.post("http://localhost:8765/v1/mcp/execute").mock(
        side_effect=_make_dm20_handler(rng)
    )

    # Shared persistence stack — ONE WriterQueue for all 4 channels.
    db_path = str(tmp_path / "stress.sqlite3")
    await bootstrap(db_path)
    wq = WriterQueue(db_path)
    await wq.start()
    channel_repo = ChannelSessionRepo(db_path, wq)
    view_repo = PersistentViewRepo(db_path, wq)

    # Shared MCP client + cache.
    client = MCPClient("http://localhost:8765")
    cache = MCPCache(client)

    # Shared monster memory.
    registry = MonsterMemoryRegistry()

    # Worker accounting.
    errors: list[str] = []
    bypass_count: dict[str, int] = {}
    cacheable_count: dict[str, int] = {}

    channels = [
        (f"ch-{i}", [f"pc-{i}-{j}" for j in range(PCS_PER_CHANNEL)])
        for i in range(NUM_CHANNELS)
    ]

    t0 = time.monotonic()
    try:
        await asyncio.gather(*(
            _channel_worker(
                cid, pcs, cache, wq, channel_repo, view_repo, registry,
                errors, bypass_count, cacheable_count,
            )
            for cid, pcs in channels
        ))
        wallclock_s = time.monotonic() - t0

        # Concurrent READ path while writes are still possible (exercises WAL).
        async with open_connection(db_path) as ro_conn:
            cur = await ro_conn.execute("SELECT COUNT(*) FROM persistent_views")
            row = await cur.fetchone()
            assert row is not None
            views_written = int(row[0])
        # 4 channels × 5 rounds = 20 view inserts expected.
        assert views_written == NUM_CHANNELS * ROUNDS_PER_CHANNEL, (
            f"View row loss: {views_written} != {NUM_CHANNELS * ROUNDS_PER_CHANNEL}"
        )

        # — ASSERTION (a): zero "database is locked" or any worker error —
        assert errors == [], f"Worker errors: {errors}"

        # — ASSERTION (b): per-channel state isolation in MonsterMemory —
        for cid, pcs in channels:
            mem = registry.recall(cid, "stress-session", "monster-x")
            seen = set(mem.damage_dealt_by.keys())
            assert seen == set(pcs), (
                f"Cross-channel leak in {cid}: expected {pcs}, got {sorted(seen)}"
            )
            # And conversely — pcs from other channels MUST NOT appear in this channel's memory.
            for other_cid, other_pcs in channels:
                if other_cid == cid:
                    continue
                for other_pc in other_pcs:
                    assert other_pc not in mem.damage_dealt_by, (
                        f"Leak: {other_pc} (from {other_cid}) appeared in {cid}'s memory"
                    )

        # — ASSERTION (c): SmartMonsterDriver-style per-round cache keys —
        # The shape under test is the cache KEY structure
        # `(channel_id, round_number, monster_id)`; we mirror it here without
        # spinning up the full LLM oracle (that's covered by Phase 10 tests).
        # Verifies channel_id is the discriminator so 4 channels with identical
        # (round, monster) do NOT collide.
        from collections import OrderedDict
        from eldritch_dm.gameplay.smart_monster_driver import _DEFAULT_CACHE_MAX_SIZE  # noqa: F401
        smart_cache: OrderedDict[tuple[str, int, str], str] = OrderedDict()
        for cid, _pcs in channels:
            for round_no in range(ROUNDS_PER_CHANNEL):
                key = (cid, round_no, "monster-x")
                # If channel_id were dropped from the key, this would overwrite.
                assert key not in smart_cache
                smart_cache[key] = f"target-for-{cid}-r{round_no}"
        assert len(smart_cache) == NUM_CHANNELS * ROUNDS_PER_CHANNEL

        # — ASSERTION (d): MCPCache L1 hit/miss internal consistency —
        # Total cacheable calls across all channels:
        total_cacheable = sum(cacheable_count.values())
        total_bypass = sum(bypass_count.values())
        assert total_cacheable == NUM_CHANNELS * ROUNDS_PER_CHANNEL * MONSTERS_PER_ROUND
        assert total_bypass == NUM_CHANNELS * ROUNDS_PER_CHANNEL

        snap = await cache.metrics_snapshot()
        # Hits + misses across L1 = total cacheable calls (every call is either hit or miss at L1).
        assert snap.hits_l1 + snap.misses_l1 == total_cacheable, (
            f"L1 accounting mismatch: hits={snap.hits_l1} misses={snap.misses_l1} expected={total_cacheable}"
        )
        # Bypass counter must match exactly the non-cacheable combat calls.
        assert snap.bypass_count == total_bypass, (
            f"Bypass mismatch: {snap.bypass_count} != {total_bypass}"
        )
        # We issued exactly 2 distinct cacheable arg-sets → at most 2 L1 misses.
        # All remaining cacheable calls should be hits (modulo TTL — 300s default,
        # well above our <60s wallclock).
        assert snap.misses_l1 <= 2, f"Too many L1 misses: {snap.misses_l1}"
        # — D-195(d) extension: L2 internal consistency — sizes are non-negative
        # and L2 received at most the same number of writes as L1 misses (the
        # cache populates L2 only on full miss).
        assert snap.size_l2 >= 0, "L2 disabled or sentinel leaked into snapshot"
        assert snap.size_l2 <= snap.misses_l1, (
            f"L2 size {snap.size_l2} exceeds L1 misses {snap.misses_l1}"
        )

        # — ASSERTION (e): WriterQueue.stop() cleanly drains in-flight writes —
        # We've already awaited all workers — queue should be empty before stop().
        # The clean-shutdown contract is: stop() completes without raising and
        # subsequent submit() raises RuntimeError.
        assert wq.qsize() == 0
        await wq.stop()
        with pytest.raises(RuntimeError):
            await channel_repo.upsert(channel_id="post-stop", campaign_name="X")

        # Wallclock budget (D-200 / success criterion 4).
        assert wallclock_s <= WALLCLOCK_BUDGET_S, (
            f"Wallclock {wallclock_s:.1f}s exceeded {WALLCLOCK_BUDGET_S}s budget"
        )
    finally:
        # Defensive cleanup — stop() is idempotent enough; aclose cache+client.
        await cache.aclose(aclose_inner=True)
```
